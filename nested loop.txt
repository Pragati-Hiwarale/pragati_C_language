 // Multilplication table 
 
#include <stdio.h>

int main()
{
    int a,b;
    
    for(a=2;a<=10;a++){
        for(b=1;b<=10;b++){
            printf("%d ",b*a);
        }
        printf("\n");
    }
    return 0;
}

#include <stdio.h>

int main()
{
    int a;
    printf("Enter number of Rows = ");
    scanf("%d",&a);
    
    for(int b=1;b<=a;b++){
        for(int c=1;c<=b;c++){
            printf("*");
        }
        printf("\n");
    }
    return 0;
}
