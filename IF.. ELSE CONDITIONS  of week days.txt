//IF.. ELSE CONDITIONS  of week days

#include <stdio.h>

int main()
{
    int num;
    printf("Enter any number between 1 to 7 %d \n");
    scanf("%d",&num);
    
    if (num ==1){
        printf("It's a Monday.");
    }
    else if(num==2){
        printf("It's a Tuesday");
    }
    else if(num==3){
        printf("It is Wednesday.");
    }
    else if(num == 4){
        printf("Heyy there It was Thrusday.");
    }
    else if(num==5){
        printf("Hello It is Friday.");
    }
    else if (num == 6){
        printf(" hello Its Saturday weekend");
    }
    else if(num==7){
        printf(" Its a Happy weekend , Sunday.");
    }
    else{
        printf(" Sorry we couldn't find that, Try again.");
    }
    return 0;
}
