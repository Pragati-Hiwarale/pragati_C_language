#include <stdio.h>
//Addition with two values
int main ()
{
    int a = 100, b = 120;
    printf("%d \n", a+b);
    printf("%d + %d = %d",a,b, a + b);
    return 0;
}

#include <stdio.h>
//Addition with three values
int main ()
{
    int a = 10, b = 20, c= 40;
    printf("%d \n", a+b+c);
    printf("%d + %d + %d = %d",a,b,c, a + b + c);
    return 0;
}

#include <stdio.h>
//Addition with Four values
int main ()
{
    int a = 1070, b = 2120, c= 312,d= 4120;
    printf("%d \n", a+b+c+d);
    printf("%d + %d + %d +%d = %d",a,b,c,d, a + b+ c+d);
    return 0;
}
